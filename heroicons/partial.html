{{- $variantDir := "24/outline" -}} {{- /* set variantDir based on variant */ -}} {{- with .variant -}} {{- if ne . "outline" "solid" "mini" "micro" -}} {{- errorf "Invalid variant %q in %q" . templates.Current.Name -}} {{- else -}} {{- if eq . "outline" -}} {{- $variantDir = "24/outline" -}} {{- else if eq . "solid" -}} {{- $variantDir = "24/solid" -}} {{- else if eq . "mini" -}} {{- $variantDir = "20/solid" -}} {{- else if eq . "micro" -}} {{- $variantDir = "16/solid" -}} {{- end -}} {{- end -}} {{- $context := . -}} {{- with $icon := .icon -}} {{- $href := printf "heroicons/%s/%s.svg" $variantDir $icon -}} {{- partial "svg" (merge $context (dict "href" $href)) -}} {{- else -}} {{- errorf "No icon provided for %q" templates.Current.Name -}} {{- end -}}
